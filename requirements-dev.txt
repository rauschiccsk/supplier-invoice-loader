# ============================================================================
# USAGE EXAMPLES
# ============================================================================
# Install development dependencies:
#   pip install -r requirements-dev.txt
#
# Run tests:
#   pytest                                    # Run all tests
#   pytest tests/test_extraction.py          # Run specific test file
#   pytest -v                                 # Verbose output
#   pytest --cov=. --cov-report=html         # With coverage report
#   pytest -k "test_extraction"              # Run tests matching pattern
#
# Format code:
#   black .                                   # Format all Python files
#   black main.py                             # Format specific file
#   black --check .                           # Check without modifying
#
# Sort imports:
#   isort .                                   # Sort all imports
#   isort --check-only .                      # Check without modifying
#
# Check code style:
#   flake8 .                                  # Check all files
#   flake8 main.py                            # Check specific file
#   flake8 --max-line-length=100 .           # Custom line length
#
# Type checking:
#   mypy .                                    # Check all files
#   mypy main.py                              # Check specific file
#   mypy --ignore-missing-imports .          # Ignore missing type stubs
#
# Security checks:
#   safety check                              # Check installed packages
#   safety check -r requirements.txt         # Check requirements file
#   bandit -r .                               # Security scan all files
#   bandit -r . -ll                           # Only high/medium severity
#
# Combined checks (run before commit):
#   black . && isort . && flake8 . && mypy . && pytest
#
# IPython debugging:
#   ipython                                   # Start IPython REPL
#
#   # In code, add breakpoint:
#   import ipdb; ipdb.set_trace()
#
